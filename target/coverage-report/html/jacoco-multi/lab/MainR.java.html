    public static int[] sort(int[] toBeOrdered){
        for(int i=0;i<toBeOrdered.length;i++){
            for(int j=i+1;j<toBeOrdered.length;j++){
                if(toBeOrdered[i]>toBeOrdered[j]){
                    int tmp=toBeOrdered[i];
                    toBeOrdered[i]=toBeOrdered[j];
                    toBeOrdered[j]=tmp;
                }
            }
        }
        return toBeOrdered;
    }
